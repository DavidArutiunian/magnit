Table task {
  id int
  title varchar
  stage_title varchar
  stage_until datetime
  location_region varchar
  location_branch varchar
  location_format varchar
  location_address varchar
  assignee varchar // FK к исполнителю
}

Table task_document {
  task_id int [ref: > task.id]
  template_id int [ref: > template.id]
}

Table template {
 id int [pk]
 title varchar
 description varchar
 sections json
 created_at datetime
 updated_at datetime
 type template_type
}

enum template_type {
  light
  complex
}
